```python
from scrapers.base import Scraper
from models import SupermarketResult
from typing import List
from crawl4ai import AsyncWebCrawler
from llm import extract_products_from_markdown
from config import get_settings
from urllib.parse import quote_plus

class MercadonaScraper(Scraper):
    """Mercadona supermarket scraper."""
    BASE_URL = "https://tienda.mercadona.es"
    SEARCH_PATH = "/buscar"
    
    async def search(self, query: str, limit: int = 10) -> List[SupermarketResult]:
        url = f"{self.BASE_URL}{self.SEARCH_PATH}?q={quote_plus(query)}"
        settings = get_settings()
        
        async with AsyncWebCrawler(
            headless=True,
            timeout=settings.crawl4ai_timeout,
            browser_args=[
                "--no-sandbox",
                "--disable-dev-shm-usage",
                "--user-agent=Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
            ],
        ) as crawler:
            result = await crawler.arun(url=url)
            
            if not result.success:
                raise Exception(f"Mercadona crawl failed: {result.error_message}")
            
            markdown = result.markdown
            products = await extract_products_from_markdown(markdown, query)
            
            # Extract product URLs from crawl result
            if result.links:
                internal_links = result.links.get("internal", [])
                product_links = [
                    link["href"] for link in internal_links 
                    if "/product/" in link.get("href", "")
                ]
                # Map URLs to products by position
                for i, product in enumerate(products):
                    if i < len(product_links):
                        product.url = product_links[i]
            
            return products[:limit]
```

Key implementation details:
- Use BASE_URL = "https://tienda.mercadona.es"
- SEARCH_PATH = "/buscar" 
- Add custom User-Agent to avoid bot detection
- Extract product URLs from result.links internal links
- Reuse extract_products_from_markdown (same LLM prompt works for Spanish)
